=== Compiling and uploading programs

yum install avr-gcc avr-libc avrdude

avr-gcc -Os -DF_CPU=16000000UL -mmcu=atmega328p -o blink blink.c

avr-objcopy -O ihex -R .eeprom ./blink ./blink.hex
avrdude \
    -c arduino \
    -p ATMEGA328P \
    -P /dev/ttyACM0 \
    -b 115200 \
    -U flash:w:./blink.hex:i
